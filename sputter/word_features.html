192@dataclass
193class WordFeaturePrecomputes:
194    """Precomputes for word features."""
195
196    __ORD_A = ord("A")
197
198    word_letter_counts: List[int]
199    """A list of letter counts in the word, where index 0 is A, 1 is B, etc."""
200
201    letter_bank: Set[str]
202    """The set of letters in the word."""
203
204    def __init__(self, word: str):
205        self.word_letter_counts = [0] * 26
206        for letter in word:
207            self.word_letter_counts[ord(letter) - WordFeaturePrecomputes.__ORD_A] += 1
208        self.letter_bank = set(word)
209
210    def get_letter_count(self, letter: str):
211        """Return the number of times the letter appears in the word."""
212        return self.word_letter_counts[ord(letter) - WordFeaturePrecomputes.__ORD_A]

229class LetterCountFeature(WordFeature):
230    """Word features based on number of letter occurrences."""
231
232    letter: str
233    """The letter to count. Must be uppercase."""
234
235    min_count: int
236    """The minimum number of occurrences of the letter to satisfy this feature."""
237
238    def evaluate(
239        self,
240        word: str,
241        ws: WordStatistics,
242        precomputes: Optional[WordFeaturePrecomputes],
243    ) -> bool:
244        if precomputes:
245            return precomputes.get_letter_count(self.letter) >= self.min_count
246        return word.count(self.letter) >= self.min_count
247
248    def __repr__(self) -> str:
249        return f"at least {self.min_count} occurrences of {self.letter}"

315@dataclass(frozen=True)
316class AlternatesVowelConsonantFeature(WordFeature):
317    """Word features based on alternating vowel and consonant occurrences."""
318
319    def evaluate(
320        self,
321        word: str,
322        ws: WordStatistics,
323        precomputes: Optional[WordFeaturePrecomputes],
324    ) -> bool:
325        last_was_vowel = word[0] in VOWELS
326        for c in word[1:]:
327            if c in VOWELS:
328                if last_was_vowel:
329                    return False
330            elif not last_was_vowel:
331                return False
332            last_was_vowel = c in VOWELS
333        return True
334
335    def __repr__(self) -> str:
336        return "alternates between vowels and consonants"

463class WordFeatureStatistics:
464    """A class for computing statistics about word features."""
465
466    def __init__(self, ws: Optional[WordStatistics] = None):
467        """Initialize a set of word feature statistics based on word frequencies.
468
469        :param ws: The WordStatistics to use. If None, a new WordStatistics will be
470            created.
471        """
472        self._ws = ws or WordStatistics()
473
474        self._feature_count: Dict[WordFeature, int] = defaultdict(int)
475
476        for word, freq in self._ws.word_frequencies().items():
477            precomputes = WordFeaturePrecomputes(word)
478            for feature in ALL_FEATURES:
479                if feature.evaluate(word, self._ws, precomputes):
480                    self._feature_count[feature] += freq
481
482        word_frequency_total = self._ws.word_frequency_total()
483        self._feature_log_prob = {
484            feature: math.log(count / word_frequency_total)
485            for feature, count in self._feature_count.items()
486        }
487
488    def evaluate_words(
489        self, words: List[str], top_n: Optional[int] = 10
490    ) -> List[WordFeatureResult]:
491        """Evaluate a set of words against all word features.
492
493        :param words: The words to evaluate.
494        :param top_n: The number of top results to return. If None, all results will be
495            returned.
496
497        :return: A list of WordFeatureResults, one for each word feature. The list is
498            sorted by log probability, from least likely to most likely.
499        """
500        precomputes = [WordFeaturePrecomputes(word) for word in words]
501        results = []
502        for feature, feature_log_prob in self._feature_log_prob.items():
503            evals = [
504                feature.evaluate(word, self._ws, precompute)
505                for word, precompute in zip(words, precomputes, strict=True)
506            ]
507            if any(evals):
508                satisfied_words = [
509                    word for word, e in zip(words, evals, strict=True) if e
510                ]
511                results.append(
512                    WordFeatureResult(
513                        feature,
514                        satisfied_words,
515                        feature_log_prob * len(satisfied_words),
516                    )
517                )
518        results.sort(key=lambda result: result.log_prob)
519        if top_n is not None:
520            results = results[:top_n]
521        return results

496            returned.
